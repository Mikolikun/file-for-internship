#include <Wire.h>
#include <BH1750.h>

BH1750 lightMeter;

void setup(){
  Serial.begin(115200);
  Wire.begin(D2,D1);
  lightMeter.begin();
}	

void loop(){
  float lux = lightMeter.readLightLevel();
  Serial.print("Light : ");
  Serial.print(lux);
  Serial.print("lu");
  delay(1000);
}
