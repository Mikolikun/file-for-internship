#include "DHT.h"

#define DHTPIN D4
#define DHTTYPE DHT22

DHT dht(DHTPIN,DHTTYPE);

void setup(){
  Serial.begin(115200);
  dht.begin();
}

void loop(){
  float h = dht.readHumidity();
  float t = dht.readTemperature();
  float f = dht.readTemperature(true);

  Serial.print("Humid : ");
  Serial.println(h);
  Serial.print("Temp : ");
  Serial.print(t);
  Serial.print(" °C ");
  Serial.print(f);
  Serial.print(" °F ");
  delay(1000);
}
